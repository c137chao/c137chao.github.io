2.pingpong

在user目录下增加一个pingpong.c, 完成进程之间的通信，子进程接收到消息打印ping，父进程收到消息打印pong, 此实验主要学习管道的使用，是第三个作业primes的基础

  • Use pipe to create a pipe.
  • Use fork to create a child.
  • Use read to read from the pipe, and write to write to the pipe.
  • Use getpid to find the process ID of the calling process.
  • Add the program to UPROGS in Makefile.
  • User programs on xv6 have a limited set of library functions available to them. You can see the list in user/user.h; the source (other than for system calls) is in user/ulib.c, user/printf.c, and user/umalloc.c.
// user/pingpong.c
#include "kernel/types.h"
#include "user/user.h"
//#include <stddef.h>

#define READ_PORT  0
#define WRITE_PORT 1

int main(int argc, char * argv[]){
    if(argc > 1){
        fprintf(2, "error:unvalid  parameters\n"); 
        exit(1);
    }     
        
    char buf[512];          // buffer between child and parent
    int fd1[2], fd2[2];

    pipe(fd1),  pipe(fd2);  // create pipe
    int pid = fork();       // create child

    if(pid == 0){               // child process
        close(fd1[WRITE_PORT]); // read  from fd1
        close(fd2[READ_PORT]);  // write from fd2

        // child recevice message
        if(read(fd1[READ_PORT], buf, 1) == 1){ 
            printf("%d: received ping\n", getpid());  // if use some write to complete, sometime occurs disorder, why?
        }
        // child write   message
        char *c_msg = "Send a massage to parent process\n";
        write(fd2[WRITE_PORT], c_msg, strlen(c_msg));
        close(fd2[WRITE_PORT]);

    }else{                 // parent process
        close(fd1[READ_PORT]);
        close(fd2[WRITE_PORT]);
        char *p_msg = "Send a massage to child  process\n"; 
        write(fd1[WRITE_PORT], p_msg, strlen(p_msg));
        wait(0);

        if(read(fd2[READ_PORT], buf, 1) == 1){
            close(fd1[WRITE_PORT]);
            //printf("<%d>: received pong\n", getpid()); 
            printf("%d: received pong\n", getpid()); 
            //write(1, buf, strlen(buf));
        }
        // else out error info and exit
        
        close(fd2[READ_PORT]);
    }
    exit(0); // should not forget exit
}

3. primes

这个作业我觉得lab1里最有意思的一个，第一次看到多进程去筛选素数，非常新奇.

  • Be careful to close file descriptors that a process doesn’t need, because otherwise your program will run xv6 out of resources before the first process reaches 35.
  • Once the first process reaches 35, it should wait until the entire pipeline terminates, including all children, grandchildren, &c. Thus the main primes process should only exit after all the output has been printed, and after all the other primes processes have exited.
  • Hint: read returns zero when the write-side of a pipe is closed.
  • It’s simplest to directly write 32-bit (4-byte) ints to the pipes, rather than using formatted ASCII I/O.
  • You should create the processes in the pipeline only as they are needed.
  • Add the program to UPROGS in Makefile.
// user/primes.c
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

int Fork() {
    int pid = fork();
    if (pid < 0) {
        fprintf(2, "fork failed\n");
        exit(-1);
    }
    return pid;
}
  
void pipe_execute_parallel(int fd_in) {
    int base;
    int number;

    if (read(fd_in, &base, sizeof(int)) != sizeof(int)) {
        // no more numbers
         exit(0);
    }
    printf("prime %d\n", base);

    int fd[2];
    pipe(fd);

    int pid = Fork()
  
    if (pid == 0) {
        close(fd[0]);
        
        while (read(fd_in, &number, sizeof(int)) > 0) {
            if (number % base != 0) {
                write(fd[1], &number, sizeof(int));
            }
        }
        close(fd[1]);
        close(fd_in);

        exit(0);
        
    } else {
        close(fd[1]);
        close(fd_in);
        pipe_execute_parallel(fd[0]);
    }

}
  
int main(int argc, char* argv[]) {
    // default max number is 35
    int max_number = 35;

    // if user pass a arguement, set max number to it
    if (argc == 2) {
        max_number = atoi(argv[1]);
    }

    int fd[2];
    pipe(fd);

    int pid = Fork();

    if (pid == 0) {
        close(fd[1]);
        pipe_execute_parallel(fd[0]);
    } else {
        close (fd[0]);
        for (int i = 2; i <= max_number; i++) {
            write(fd[1], &i, sizeof(int));
        }
        close (fd[1]);
        wait(0);
    }
    exit(0);

}

4.find

在user下添加find.c，搜索某个目录下文件名字匹配的文件，基本仿照user/ls.c做搜索，注意细节上做一些修改，参照user/grep.c中的匹配方式，让find查找为支持正则匹配(这个是挑战里面的内容) 因为要用到的代码基本都在ls.c和grep.c中写好了，所以实际上基本不用自己写什么代码

// user/find.c
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"
#include "kernel/fs.h"

char*
fmate_name(char *path)  
{
  static char buf[DIRSIZ+1];
  char *p;

  // Find first character after last slash.
  for(p=path+strlen(path); p >= path && *p != '/'; p--)
    ;
  p++;

  // Return blank-padded name.
  if(strlen(p) >= DIRSIZ)
    return p;
  memmove(buf, p, strlen(p));
  // memset(buf+strlen(p), ' ', DIRSIZ-strlen(p));  // be careful, don't append space
  return buf;
}

int matchhere(char*, char*);
int matchstar(int, char*, char*);

int
match(char *re, char *text)
{
  if(re[0] == '^')
    return matchhere(re+1, text);
  do{  // must look at empty string
    if(matchhere(re, text))
      return 1;
  }while(*text++ != '\0');
  return 0;
}

// matchhere: search for re at beginning of text
int matchhere(char *re, char *text)
{
  if(re[0] == '\0')
    return 1;
  if(re[1] == '*')
    return matchstar(re[0], re+2, text);
  if(re[0] == '$' && re[1] == '\0')
    return *text == '\0';
  if(*text!='\0' && (re[0]=='.' || re[0]==*text))
    return matchhere(re+1, text+1);
  return 0;
}

// matchstar: search for c*re at beginning of text
int matchstar(int c, char *re, char *text)
{
  do{  // a * matches zero or more instances
    if(matchhere(re, text))
      return 1;
  }while(*text!='\0' && (*text++==c || c=='.'));
  return 0;
}

void
find(char *path, char * target)
{
  char buf[512], *p;
  int fd;
  struct dirent de;
  struct stat st;

  if((fd = open(path, 0)) < 0){
    fprintf(2, "ls: cannot open %s\n", path);
    return;
  }

  if(fstat(fd, &st) < 0){
    fprintf(2, "ls: cannot stat %s\n", path);
    close(fd);
    return;
  }

  switch(st.type){
  case T_FILE:
    //printf("%s . %s, %d\n", fmate_name(path), target, strcmp(fmate_name(path), target));
    if( match(target, fmate_name(path)) )   printf("%s\n", path);
    break;

  case T_DIR:
    if(strlen(path) + 1 + DIRSIZ + 1 > sizeof buf){
      printf("ls: path too long\n");
      break;
    }
    strcpy(buf, path);
    p = buf+strlen(buf);
    *p++ = '/';
    
    while(read(fd, &de, sizeof(de)) == sizeof(de)){
      if(de.inum == 0 || !strcmp(de.name, ".") || !strcmp(de.name, ".."))
        continue;
      memmove(p, de.name, DIRSIZ);
      p[DIRSIZ] = 0;
      if(stat(buf, &st) < 0){
        printf("ls: cannot stat %s\n", buf);
        continue;
      }
      find(buf, target);
      //printf("%s %d %d %d\n", fmtname(buf), st.type, st.ino, st.size);
    }
    break;
  }
  close(fd);
}


int main(int argc, char* argv[]){
    // check argv
    if(argc != 3){
        fprintf(2, "try: find search_direct search_filename\n");
        exit(1);
    }
    find(argv[1], argv[2]);
    exit(0); 
}

5.xargs

简单来说xargs就是从stdin中读取参数，每一行都作为argv去exec一次

  • Use fork and exec to invoke the command on each line of input. Use wait in the parent to wait for the child to complete the command.
  • To read individual lines of input, read a character at a time until a newline ('\n') appears.
  • kernel/param.h declares MAXARG, which may be useful if you need to declare an argv array.
  • Add the program to UPROGS in Makefile.
  • Changes to the file system persist across runs of qemu; to get a clean file system run make clean and then make qemu.
// user/xargs.c
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"
#include "kernel/fs.h"
#include "kernel/param.h"
#include <stddef.h>

#define BUFFSZ 128

/**
    ider: example echo hello | xargs echo bye

     1.read the result from stdin 
     2.for every line: 
            product the new argv
            exec("echo", new argv) in child; 

**/
int Fork(){
    int pid;
    if((pid = fork()) < 0){
        printf("error fork failed\n");
        exit(-1);
    }
    return pid;
}

int main(int argc, char *argv[]) {
    if(argc + 1 > MAXARG){
      fprintf(2, "too much paremeters\n");
      exit(-1);
    }
    int end = 0;
    // char *p = buf;
    // read argvs from pipe
    while(1){
        char buf[BUFFSZ]; 
        if(read(0, &buf[end], 1) != 1) 
             break;
        if(buf[end] == '\n'){
            buf[end] = '\0';
            printf("%s:", buf);
            //strcpy(argv[argc] , buf);
            argv[argc] = buf; 
            argv[argc + 1] = NULL;
            // exec comd
            if(Fork() == 0){
                exec(argv[1], &argv[1]);
                fprintf(2,"file %s didn't exist\n", argv[0]);
                exit(-1);
            }else{
                wait(0);
                end = -1;
            }
            
        }
        ++end;
    }
    
    exit(0);
}
