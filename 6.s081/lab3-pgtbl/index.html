1.Print a page table

主要要完成的就是在kernel/vm.c中定义vmprint(pagetable_t), 用来打印页表，有多种方法可以实现，用多层for循环比较直观，也可以参照freewalk（CV大法好），使用递归完成，是一个热身作业。

  • You can put vmprint() in kernel/vm.c.
  • Use the macros at the end of the file kernel/riscv.h.
  • The function freewalk may be inspirational.
  • Define the prototype for vmprint in kernel/defs.h so that you can call it from exec.c.
  • Use %p in your printf calls to print out full 64-bit hex PTEs and addresses as shown in the example.
void vmprint_recur(pagetable_t pagetable, int level){
// 有很多方式处理.. ..的打印
  for(int i = 0; i < 512; i++){
    pte_t pte = pagetable[i];
    if((pte & PTE_V) && (pte & (PTE_R|PTE_W|PTE_X)) == 0){
      // this PTE points to a lower-level page table.
      uint64 nextlevelPT = PTE2PA(pte);
      switch(level){
        case 0:   printf("..");     break;
        case 1:   printf(".. ..");  break;
        default:  break;
      }
      printf("%d: pte %p pa %p\n", i, pte, nextlevelPT);
      vmprint_recur((pagetable_t)nextlevelPT, level + 1);
      
    } else if(pte & PTE_V){ 
      // this PTE points to the last level page table
      uint64 pa = PTE2PA(pte);
      printf(".. .. ..%d: pte %p pa %p\n", i, pte, pa);
    }
  }
  return;
}

void vmprint(pagetable_t pte){  // pagetable_t is uint64*
  // print pagetables
  printf("page table %p\n", pte);
  vmprint_recur(pte, 0); 
  return;   
}

2.kernel page table per proc

task2 和 task3 的难度很大，尤其是task3，我在做task3的时候被各种panic折磨崩溃

task2目的是为每一个进程添加一个内核页表，原本所有进程进入内核是共用一个大页表 kernel-pagetable(看kvminit的代码就可以知道，xv6book 3.2也提到了)，这个kernel-pagetable使用直接映射(虚拟地址就是物理地址)，并且为每一个进程的内核栈都预留出了一个guard page，现在要为每个进程添加一个自己的内核页表并相应修改内核栈的申请与释放。

在task3提到一个要求 (task2没有要求，不会影响task2的通过)，这里就提前剧透了：

为了让内核能够直接访问用户页面的虚拟地址(原本是先用户页表翻译成物理地址，然后再传给内核，这样会产生额外的开销，这里到task3我会再去解释一下)，那最好将user pagetable的内容同步到kernel pagetable中去，这样就需要保证用户空间和内核空间的虚拟地址不会重复，xv6的做法是令user spcae的虚拟地址是从0开始增长，而从PLIC(0x0c000000)开始是kernel space的地址，这样就需要限制用user space的虚拟地址不会超过PLIC，可以看到CLINT(Core Local Interruptor)比PLIC要小，并且CLINT只有在内核初始化的时候需要，所以进程内核栈不需要映射CLINT。

(1)按照xv6book里所提到的，我们添加一个kvmake辅助内核页表的初始化工作，kvmmap的参数做一点修改(记得在def.h里添加/修改声明)

pagetable_t kvmake(){
  pagetable_t pg = (pagetable_t) kalloc();
  memset(kernel_pagetable, 0, PGSIZE);
  
  // uart registers
  kvmmap(pg, UART0, UART0, PGSIZE, PTE_R | PTE_W);
  // virtio mmio disk interface
  kvmmap(pg, VIRTIO0, VIRTIO0, PGSIZE, PTE_R | PTE_W);
  // PLIC
  kvmmap(pg, PLIC, PLIC, 0x400000, PTE_R | PTE_W);
  // map kernel text executable and read-only.
  kvmmap(pg, KERNBASE, KERNBASE, (uint64)etext-KERNBASE, PTE_R | PTE_X);
  // map kernel data and the physical RAM we'll make use of.
  kvmmap(pg, (uint64)etext, (uint64)etext, PHYSTOP-(uint64)etext, PTE_R | PTE_W);
  // map the trampoline for trap entry/exit to
  // the highest virtual address in the kernel.
  kvmmap(pg, TRAMPOLINE, (uint64)trampoline, PGSIZE, PTE_R | PTE_X);
  
  return pg;
}

void
kvminit()
{
  kernel_pagetable = kvmake();
  // CLINT
  kvmmap(kernel_pagetable, CLINT, 0x10000, PTE_R | PTE_W);
}

// add a mapping to the kernel page table.
// only used when booting.
// does not flush TLB or enable paging.
void
kvmmap(pagetable_t pg, uint64 va, uint64 pa, uint64 sz, int perm)
{
  if(mappages(pg, va, sz, pa, perm) != 0)
    panic("kvmmap");
}

释放内核页表，task2里比较关键的一部分，除了kernel stack之外，kernel page table里管理的其他内存应该都是公共的I/O地址(只是猜测^.^)，I/O设备内存不可以释放，这里有两种方法:

  • 先unmap掉kvmmake里建立的I/O映射，然后直接freewalk;
  • 也可以只释放页表，不释放leaf pagetable指向的内存，内核栈单独释放掉

推荐第二种做法，不然到了task3，方法(1)还要额外unmap用户空间的地址, 很容易遇到问题。

static void

void
proc_freekernpgtbl(pagetable_t pagetable, uint64 sz)
{
  // there are 2^9 = 512 PTEs in a page table.
  for(int i = 0; i < 512; i++){
    pte_t pte = pagetable[i];
    if((pte & PTE_V) && (pte & (PTE_R|PTE_W|PTE_X)) == 0){
      // this PTE points to a lower-level page table.
      uint64 child = PTE2PA(pte);
      proc_freekernpgtbl((pagetable_t)child, sz);
      pagetable[i] = 0;
    } else if(pte & PTE_V){
      pagetable[i] = 0;
    }
  }
  kfree((void*)pagetable);
}

3.Simplify copyin/copyinstr

copyin是向内核空间传递地址，原本的copyin会先在进程的用户页表walk(软件的方式)到对应的页表项，获取物理地址，然后内核拿到物理地址后，再通过MMU(硬件的方式)重新翻译一次地址(内核页表虚拟地址就是物理地址)

task3要求我们将用户页表的内容同步到内核页表中去，这样减小每次在内核访问用户地址的额外开销;

(1)用copyin_new 和 copyinstr_new替换

copyin(pagetable_t pagetable, char *dst, uint64 srcva, uint64 len)
{
  return copyin_new(pagetable, dst, srcva, len);
}

int
copyinstr(pagetable_t pagetable, char *dst, uint64 srcva, uint64 max)
{
  return copyinstr_new(pagetable, dst, srcva, max);
}

试着make qemu，panic !!!

因为现在的内核页表不含用户页表的内容，仿照uvmcopy,增加一个用户页表复制到内核页表的函数kvmcopy，为了提高效率，这里最好不要从0开始复制，不然页表中每添加一个pte都要复制整个页表，设置要复制的内存区间[beg, end), 然后把区间内的每个页面复制到内核页表中去

int
kvmcopy(pagetable_t old, pagetable_t new, uint64 beg, uint64 end){
  pte_t *pte;
  uint64 pa, i;
  uint flags;

  for(i = beg; i < end; i += PGSIZE){
    if((pte = walk(old, i, 0)) == 0)
      panic("kvmcopy: pte should exist");
    if((*pte & PTE_V) == 0)
      panic("kvmcopy: page not present");
    pa = PTE2PA(*pte);
    flags = PTE_FLAGS(*pte) & (~PTE_U); // 注意不要忘记这里，否则kerntrap
    if(copy_mappages(new, i, PGSIZE, pa, flags) != 0){
    // copy_mappages与mappages的唯一不同是copy_mappages会覆盖已经存在的页表项
    // 如果还想使用原来的mappages，在exec里就要记得对原本的内核页表做清理工作，否则会remap
      goto err;
    }
  }
  return 0;

 err:
  uvmunmap(new, 0, i / PGSIZE, 1);
  return -1;
}

修改growproc (sys_sbrk调用了growproc)

int
growproc(int n)
{
  uint sz;
  struct proc *p = myproc();

  sz = p->sz;
  if(n > 0){
    if(sz + n >= PLIC || (sz = uvmalloc(p->pagetable, sz, sz + n)) == 0) {
      return -1;
    }
    kvmcopy(p->pagetable, p->kernpgtbl, PGROUNDUP(p->sz), sz);
  } else if(n < 0){
    sz = uvmdealloc(p->pagetable, sz, sz + n);
    uvmunmap(p->kernpgtbl, PGROUNDUP(sz), -n / PGSIZE, 0); // note : n < 0

  }
  p->sz = sz;
  return 0;
}
